async function fetchPrediction() { try { const response = await fetch("https://ladder-proxy-production.up.railway.app/predict", { headers: { 'Accept': 'application/json' } }); const data = await response.json(); // ✅ 이게 핵심 console.log("✅ JSON 파싱 결과:", data); document.getElementById("round").innerText = data.predict_round ? `🎯 예측 회차: ${data.predict_round}회차` : "⛔ 예측 회차 없음"; // 아래 생략 (front_predictions, back_predictions 처리 등) } catch (error) { console.error("❌ 예측 데이터 보기 오류", error); } }
